## 问题描述
有 N 个网络节点，标记为 1 到 N。

给定一个列表 times，表示信号经过有向边的传递时间。 times[i] = (u, v, w)，其中 u 是源节点，v 是目标节点， w 是一个信号从源节点传递到目标节点的时间。

现在，我们向当前的节点 K 发送了一个信号。需要多久才能使所有节点都收到信号？如果不能使所有节点收到信号，返回 -1。

## 算法思路

计算从K到其他所有点的最短路径，用dijkstra算法就行。如果存在到不了的点，返回-1，否则返回最大的距离。

## 代码
```c++
class Solution {
public:
    vector<int> dijk(vector<vector<pair<int, int>>>& adj, int K){
        vector<int> dist(adj.size(), 1000000);
        vector<bool> find(adj.size(), false);
        dist[K] = 0;
        int cur = K;
        find[cur] = true;
        int N = int(adj.size())-1;
        for(auto& p : adj[K]){
            dist[p.first] = p.second;
        }
        for(int i = 1; i < N; i++){
            int t = 1000000;
            int n = i;
            for(int j = 1; j <= N;j++){
                if(!find[j] && dist[j] < t){
                    t = dist[j];
                    n = j;
                }
            }
            find[n] = true;
            
            for(auto& p : adj[n]){
                if(dist[n] + p.second < dist[p.first]){
                    dist[p.first] = dist[n]+p.second;
                }
            }
        }
        for(int i = 1; i <= N; i++)
            if(dist[i] == 1000000){
                //cout << i << endl;
                dist[0] = -1;
                break;
            }
        return dist; 
    }
    
    int networkDelayTime(vector<vector<int>>& times, int N, int K) {
        vector<vector<pair<int, int>>> adj(N+1, vector<pair<int, int>>());
        int res = 0;
        for(int i = 0; i < times.size(); i++){
            adj[times[i][0]].push_back(make_pair(times[i][1], times[i][2]));
        }
        vector<int> dist = dijk(adj, K);
        if(dist[0] == -1) return -1;
        for(int i=1; i <= N; i++)
            res = max(res, dist[i]);
        return res;
        
    }
};
```
